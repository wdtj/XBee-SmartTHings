027 * @see ZigBeeNetwork

031        /**
032         * Instantiates a new DigiPoint Network object.

035         * 
036         * @throws NullPointerException if {@code device == null}.
037         * 
038         * @see DigiPointDevice
039         */
040        DigiPointNetwork(DigiPointDevice device) {
041                super(device);
042        }
043}
