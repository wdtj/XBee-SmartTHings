016package com.digi.xbee.api.connection.serial;
017
018/**
019 * Helper class used to store serial connection parameters information.
020 * 
021 * <p>Parameters are stored as public variables so that they can be accessed
022 * and read from any class.</p>
023 */
024public final class SerialPortParameters {
025        
026        // Constants.
027        private static final int HASH_SEED = 23;
028        
029        // Variables.
030        public final int baudrate;
031        public final int dataBits;
032        public final int stopBits;
033        public final int parity;
034        public final int flowControl;
035        
036        /**
037         * Class constructor. Instances a new {@code SerialPortParameters} object
038         * with the given parameters.
039         * 
040         * @param baudrate Serial connection baud rate,
041         * @param dataBits Serial connection data bits.
042         * @param stopBits Serial connection stop bits.
043         * @param parity Serial connection parity.
044         * @param flowControl Serial connection flow control.
045         * 
046         * @throws IllegalArgumentException if {@code baudrate < 0} or
047         *                                  if {@code dataBits < 0} or
048         *                                  if {@code stopBits < 0} or
049         *                                  if {@code parity < 0} or
050         *                                  if {@code flowControl < 0}.
051         */
052        public SerialPortParameters(int baudrate, int dataBits, int stopBits, int parity, int flowControl) {
053                if (baudrate < 0)
054                        throw new IllegalArgumentException("Baudrate cannot be less than 0.");
055                if (dataBits < 0)
056                        throw new IllegalArgumentException("Number of data bits cannot be less than 0.");
057                if (stopBits < 0)
058                        throw new IllegalArgumentException("Number of stop bits cannot be less than 0.");
059                if (parity < 0)
060                        throw new IllegalArgumentException("Illegal parity value.");
061                if (flowControl < 0)
062                        throw new IllegalArgumentException("Illegal flow control value.");
063                
064                this.baudrate = baudrate;
065                this.dataBits = dataBits;
066                this.stopBits = stopBits;
067                this.parity = parity;
068                this.flowControl = flowControl;
069        }
070        
071        /*
072         * (non-Javadoc)
073         * @see java.lang.Object#equals(java.lang.Object)
074         */
075        @Override
076        public boolean equals(Object obj) {
077                if (obj instanceof SerialPortParameters)
078                        return ((SerialPortParameters)obj).baudrate == baudrate 
079                                && ((SerialPortParameters)obj).dataBits == dataBits 
080                                && ((SerialPortParameters)obj).stopBits == stopBits
081                                && ((SerialPortParameters)obj).parity == parity
082                                && ((SerialPortParameters)obj).flowControl == flowControl;
083                else
084                        return false;
085        }
086        
087        /*
088         * (non-Javadoc)
089         * @see java.lang.Object#hashCode()
090         */
091        @Override
092        public int hashCode() {
093                int hash = HASH_SEED;
094                hash = hash * (hash + baudrate);
095                hash = hash * (hash + dataBits);
096                hash = hash * (hash + stopBits);
097                hash = hash * (hash + parity);
098                hash = hash * (hash + flowControl);
099                return hash;
100        }
101        
102        /*
103         * (non-Javadoc)
104         * @see java.lang.Object#toString()
105         */
106        @Override
107        public String toString() {
108                return "Baud Rate: "+ baudrate + ", Data Bits: " + dataBits 
109                                + ", Stop Bits: " + stopBits + ", Parity: " + parity 
110                                + ", Flow Control: " + flowControl;
111        }
112}
