016package com.digi.xbee.api.exceptions;
017
018/**
019 * This exception will be thrown when trying to perform any operation with 
020 * the XBee device and its communication interface is closed.
021 * 
022 * @see RuntimeException
023 */
024public class InterfaceNotOpenException extends RuntimeException {
025
026        // Constants
027        private static final long serialVersionUID = 1L;
028        private static final String DEFAULT_MESSAGE = "The connection interface is not open.";
029
030        /**
031         * Creates an {@code InterfaceNotOpenException} with 
032         * {@value #DEFAULT_MESSAGE} as its error detail message.
033         */
034        public InterfaceNotOpenException() {
035                super(DEFAULT_MESSAGE);
036        }
037        
038        /**
039         * Creates an {@code InterfaceNotOpenException} with the specified 
040         * message.
041         * 
042         * @param message The associated message.
043         */
044        public InterfaceNotOpenException(String message) {
045                super(message);
046        }
047        
048        /**
049         * Creates an {@code InterfaceNotOpenException} with the specified message
050         * and cause.
051         * 
052         * @param message The associated message.
053         * @param cause The cause of this exception.
054         * 
055         * @see Throwable
056         */
057        public InterfaceNotOpenException(String message, Throwable cause) {
058                super(message, cause);
059        }
060}
