016package com.digi.xbee.api.listeners;
017
018import com.digi.xbee.api.RemoteXBeeDevice;
019
020/**
021 * Interface defining the required methods that an object should implement to be 
022 * notified about device discovery events. 
023 */
024public interface IDiscoveryListener {
025
026        /**
027         * Notifies that a remote device was discovered in the network.
028         * 
029         * @param discoveredDevice The discovered remote device.
030         * 
031         * @see com.digi.xbee.api.RemoteXBeeDevice
032         */
033        public void deviceDiscovered(RemoteXBeeDevice discoveredDevice);
034        
035        /**
036         * Notifies that an error occurred during the discovery process.
037         * 
038         * <p>This method is only called when an error occurs but does not cause 
039         * the process to finish.</p>
040         * 
041         * @param error The error message.
042         */
043        public void discoveryError(String error);
044        
045        /**
046         * Notifies that the discovery process has finished.
047         * 
048         * @param error The error message, or {@code null} if the process finished 
049         *              successfully.
050         */
051        public void discoveryFinished(String error);
052}
