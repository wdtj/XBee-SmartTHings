016package com.digi.xbee.api.listeners;
017
018import com.digi.xbee.api.models.ExplicitXBeeMessage;
019
020/**
021 * This interface defines the required methods that should be implemented to 
022 * behave as an explicit data listener and be notified when new explicit data 
023 * is received from a remote XBee device of the network.
024 */
025public interface IExplicitDataReceiveListener {
026
027        /**
028         * Called when explicit data is received from a remote node of the network.
029         * 
030         * @param explicitXBeeMessage An {@code ExplicitXBeeMessage} object 
031         *                            containing the data, the 
032         *                            {@code RemoteXBeeDevice} that sent the data, 
033         *                            a flag indicating whether the data was 
034         *                            sent via broadcast or not and the application 
035         *                            layer fields (source endpoint, destination 
036         *                            endpoint, cluster ID and profile ID).
037         * 
038         * @see com.digi.xbee.api.models.ExplicitXBeeMessage
039         * @see com.digi.xbee.api.RemoteXBeeDevice
040         */
041        public void explicitDataReceived(ExplicitXBeeMessage explicitXBeeMessage);
042}
