016package com.digi.xbee.api.listeners;
017
018import com.digi.xbee.api.RemoteXBeeDevice;
019import com.digi.xbee.api.io.IOSample;
020
021/**
022 * This interface defines the required methods that an object should implement
023 * to behave as an IO Sample listener and be notified when IO samples are 
024 * received from a remote XBee device of the network.
025 */
026public interface IIOSampleReceiveListener {
027        
028        /**
029         * Called when an IO sample is received through the connection interface.
030         * 
031         * @param remoteDevice The remote XBee device that sent the sample.
032         * @param ioSample The received IO sample.
033         * 
034         * @see com.digi.xbee.api.RemoteXBeeDevice
035         * @see com.digi.xbee.api.io.IOSample
036         */
037        public void ioSampleReceived(RemoteXBeeDevice remoteDevice, IOSample ioSample);
038}
