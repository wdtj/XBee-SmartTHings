016package com.digi.xbee.api.listeners;
017
018import com.digi.xbee.api.models.UserDataRelayMessage;
019
020/**
021 * This interface defines the required methods that should be implemented to 
022 * behave as an User Data Relay listener and be notified when new User Data
023 * Relay messages are received.
024 *
025 * @since 1.3.0
026 */
027public interface IUserDataRelayReceiveListener {
028
029        /**
030         * Called when a User Data Relay message is received.
031         *
032         * @param userDataRelayMessage A {@code UserDataRelayMessage} object
033         *                             containing the data and source interface.
034         *
035         * @see UserDataRelayMessage
036         */
037        void userDataRelayReceived(UserDataRelayMessage userDataRelayMessage);
038}
