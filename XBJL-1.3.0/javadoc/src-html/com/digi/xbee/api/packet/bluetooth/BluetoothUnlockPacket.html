016package com.digi.xbee.api.packet.bluetooth;
017
018import com.digi.xbee.api.models.SrpStep;
019import com.digi.xbee.api.packet.APIFrameType;
020import com.digi.xbee.api.packet.XBeeAPIPacket;
021import com.digi.xbee.api.utils.HexUtils;
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024
025import java.io.ByteArrayOutputStream;
026import java.io.IOException;
027import java.util.Arrays;
028import java.util.LinkedHashMap;
029
030/**
031 * This class represents a Bluetooth Unlock packet. Packet is built using the
032 * parameters of the constructor.
033 *
034 * <p>The Bluetooth Unlock packet is used to authenticate a connection on the
035 * Bluetooth interface and unlock the processing of AT command frames.</p>
036 *
037 * <p>The unlock process is an implementation of the SRP (Secure Remote
038 * Password) algorithm using the RFC5054 1024-bit group and the SHA-256 hash
039 * algorithm. The value of I is fixed to the username apiservice.</p>
040 *
041 * <p>Upon completion, each side will have derived a shared session key which is
042 * used to communicate in an encrypted fashion with the peer. Additionally, a
043 * Modem Status frame - 0x8A with the status code 0x32 (Bluetooth Connected) is
044 * sent through the UART (if AP = 1 or 2). When an unlocked connection is
045 * terminated, a Modem Status frame with the status code 0x33 (Bluetooth
046 * Disconnected) is sent through the UART.</p>
047 *
048 * @see BluetoothUnlockResponsePacket
049 * @see XBeeAPIPacket
050 *
051 * @since 1.3.0
052 */
053public class BluetoothUnlockPacket extends XBeeAPIPacket {
054
055        // Constants.
056        private static final int MIN_API_PAYLOAD_LENGTH = 34; /* 1 (Frame type) + 1 (Step) + 32 (Hash length) */
057
058        private static final String ERROR_PAYLOAD_NULL = "Bluetooth Unlock packet payload cannot be null.";
059        private static final String ERROR_INCOMPLETE_PACKET = "Incomplete Bluetooth Unlock packet.";
060        private static final String ERROR_NOT_BLE_UNLOCK = "Payload is not a Bluetooth Unlock packet.";
061        private static final String ERROR_STEP_NULL = "SRP step cannot be null.";
062        private static final String ERROR_DATA_NULL = "Data cannot be null.";
063
064        // Variables.
065        private SrpStep srpStep;
066
067        private byte[] data;
068
069        private Logger logger;
070
071        /**
072         * Creates a new {@code BluetoothUnlockPacket} object from the given
073         * payload.
074         *
075         * @param payload The API frame payload. It must start with the frame type
076         *                corresponding to a Bluetooth Unlock packet ({@code 0x2C}).
077         *                The byte array must be in {@code OperatingMode.API} mode.
078         *
079         * @return Parsed Bluetooth Unlock packet.
080         *
081         * @throws IllegalArgumentException if {@code payload[0] != APIFrameType.BLE_UNLOCK.getValue()} or
082         *                                  if {@code payload.length < }{@value #MIN_API_PAYLOAD_LENGTH}.
083         * @throws NullPointerException if {@code payload == null}.
084         */
085        public static BluetoothUnlockPacket createPacket(byte[] payload) {
086                if (payload == null)
087                        throw new NullPointerException(ERROR_PAYLOAD_NULL);
088
089                if (payload.length < MIN_API_PAYLOAD_LENGTH)
090                        throw new IllegalArgumentException(ERROR_INCOMPLETE_PACKET);
091
092                if ((payload[0] & 0xFF) != APIFrameType.BLE_UNLOCK.getValue())
093                        throw new IllegalArgumentException(ERROR_NOT_BLE_UNLOCK);
094
095                // payload[0] is the frame type.
096                int index = 1;
097
098                // SRP step byte.
099                SrpStep srpStep = SrpStep.get(payload[index] & 0xFF);
100                index = index + 1;
101
102                // Get data.
103                byte[] data = Arrays.copyOfRange(payload, index, payload.length);
104
105                return new BluetoothUnlockPacket(srpStep, data);
106        }
107
108        /**
109         * Class constructor. Instantiates a new {@code BluetoothUnlockPacket}
110         * object with the given parameters.
111         *
112         * @param srpStep The SRP step.
113         * @param data Data contained in the packet.
114         *
115         * @throws NullPointerException if {@code srpStep == null} or
116         *                              if {@code data == null}.
117         */
118        public BluetoothUnlockPacket(SrpStep srpStep, byte[] data) {
119                super(APIFrameType.BLE_UNLOCK);
120
121                if (srpStep == null)
122                        throw new NullPointerException(ERROR_STEP_NULL);
123                if (data == null)
124                        throw new NullPointerException(ERROR_DATA_NULL);
125
126                this.srpStep = srpStep;
127                this.data = data;
128                this.logger = LoggerFactory.getLogger(BluetoothUnlockPacket.class);
129        }
130
131        @Override
132        public byte[] getAPIPacketSpecificData() {
133                ByteArrayOutputStream os = new ByteArrayOutputStream();
134                try {
135                        os.write(srpStep.getID());
136                        os.write(data);
137                } catch (IOException e) {
138                        logger.error(e.getMessage(), e);
139                }
140                return os.toByteArray();
141        }
142
143        @Override
144        public boolean needsAPIFrameID() {
145                return false;
146        }
147
148        @Override
149        public boolean isBroadcast() {
150                return false;
151        }
152        
153        /**
154         * Sets the SRP step.
155         *
156         * @param srpStep The new SRP step.
157         *
158         * @throws NullPointerException if {@code srpStep == null}.
159         *
160         * @see #getSrpStep()
161         * @see SrpStep
162         */
163        public void setSrpStep(SrpStep srpStep) {
164                if (srpStep == null)
165                        throw new NullPointerException(ERROR_STEP_NULL);
166                this.srpStep = srpStep;
167        }
168        
169        /**
170         * Retrieves the SRP step.
171         *
172         * @return The SRP step.
173         *
174         * @see #setSrpStep(SrpStep)
175         * @see SrpStep
176         */
177        public SrpStep getSrpStep() {
178                return srpStep;
179        }
180        
181        /**
182         * Sets the SRP data.
183         *
184         * @param data The new SRP data.
185         *
186         * @throws NullPointerException if {@code data == null}.
187         *
188         * @see #getData()
189         */
190        public void setData(byte[] data) {
191                if (data == null)
192                        throw new NullPointerException(ERROR_DATA_NULL);
193                this.data = data;
194        }
195        
196        /**
197         * Retrieves the SRP data.
198         *
199         * @return The SRP data.
200         *
201         * @see #setData(byte[])
202         */
203        public byte[] getData() {
204                return data;
205        }
206
207        @Override
208        public LinkedHashMap<String, String> getAPIPacketParameters() {
209                LinkedHashMap<String, String> parameters = new LinkedHashMap<>();
210                parameters.put("SRP step", String.format("%s (%s)", HexUtils.prettyHexString(HexUtils.integerToHexString(
211                                srpStep.getID(), 1)), srpStep.getDescription()));
212                parameters.put("Data", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(data)));
213                return parameters;
214        }
215}
