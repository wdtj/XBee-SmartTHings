036/**
037 * This class represents an IO Data Sample RX Indicator packet. Packet is built 
038 * using the parameters of the constructor or providing a valid API payload.
039 * 
040 * <p>When the module receives an IO sample frame from a remote device, it 
041 * sends the sample out the UART using this frame type (when AO=0). Only modules
042 * running API firmware will send IO samples out the UART.</p>
043 * 
044 * <p>Among received data, some options can also be received indicating 
045 * transmission parameters.</p>
046 * 
047 * @see com.digi.xbee.api.models.XBeeReceiveOptions
048 * @see com.digi.xbee.api.packet.XBeeAPIPacket
049 */
050public class IODataSampleRxIndicatorPacket extends XBeeAPIPacket {
051
052        // Constants.
053        private static final int MIN_API_PAYLOAD_LENGTH = 12; // 1 (Frame type) + 8 (32-bit address) + 2 (16-bit address) + 1 (receive options)
054        
055        // Variables.
056        private final XBee64BitAddress sourceAddress64;
057        private final XBee16BitAddress sourceAddress16;
058        
059        private IOSample ioSample;
060        
061        private final int receiveOptions;
062        
063        private byte[] rfData;
064        
065        private Logger logger;
066        
067        /**
068         * Creates a new {@code IODataSampleRxIndicatorPacket} object from the 
069         * given payload.
070         * 
071         * @param payload The API frame payload. It must start with the frame type 
072         *                corresponding to a IO Data Sample RX Indicator packet ({@code 0x92}).
073         *                The byte array must be in {@code OperatingMode.API} mode.
074         * 
075         * @return Parsed IO Data Sample Rx Indicator packet.
076         * 
077         * @throws IllegalArgumentException if {@code payload[0] != APIFrameType.IO_DATA_SAMPLE_RX_INDICATOR.getValue()} or
078         *                                  if {@code payload.length < }{@value #MIN_API_PAYLOAD_LENGTH} or
079         *                                  if {@code receiveOptions < 0} or
080         *                                  if {@code receiveOptions > 255}.
081         * @throws NullPointerException if {@code payload == null}.
082         */
083        public static IODataSampleRxIndicatorPacket createPacket(byte[] payload) {
084                if (payload == null)
085                        throw new NullPointerException("IO Data Sample RX Indicator packet payload cannot be null.");
086                
087                // 1 (Frame type) + 8 (32-bit address) + 2 (16-bit address) + 1 (receive options)
088                if (payload.length < MIN_API_PAYLOAD_LENGTH)
089                        throw new IllegalArgumentException("Incomplete IO Data Sample RX Indicator packet.");
090                
091                if ((payload[0] & 0xFF) != APIFrameType.IO_DATA_SAMPLE_RX_INDICATOR.getValue())
092                        throw new IllegalArgumentException("Payload is not a IO Data Sample RX Indicator packet.");
093                
094                // payload[0] is the frame type.
095                int index = 1;
096                
097                // 2 bytes of 16-bit address.
098                XBee64BitAddress sourceAddress64 = new XBee64BitAddress(Arrays.copyOfRange(payload, index, index + 8));
099                index = index + 8;
100                
101                // 2 bytes of 16-bit address.
102                XBee16BitAddress sourceAddress16 = new XBee16BitAddress(payload[index] & 0xFF, payload[index + 1] & 0xFF);
103                index = index + 2;
104                
105                // Receive options
106                int receiveOptions = payload[index] & 0xFF;
107                index = index + 1;
108                
109                // Get data.
110                byte[] data = null;
111                if (index < payload.length)
112                        data = Arrays.copyOfRange(payload, index, payload.length);
113                
114                return new IODataSampleRxIndicatorPacket(sourceAddress64, sourceAddress16, receiveOptions, data);
115        }
116        
117        /**
118         * Class constructor. Instantiates a new 
119         * {@code IODataSampleRxIndicatorPacket} object with the given parameters.
120         * 
121         * @param sourceAddress64 64-bit address of the sender.
122         * @param sourceAddress16 16-bit address of the sender.
123         * @param receiveOptions Receive options.
124         * @param rfData Received RF data.
125         * 
126         * @throws IllegalArgumentException if {@code receiveOptions < 0} or
127         *                                  if {@code receiveOptions > 255}.
128         * @throws NullPointerException if {@code sourceAddress64 == null} or 
129         *                              if {@code sourceAddress16 == null}.
130         * 
131         * @see com.digi.xbee.api.models.XBeeReceiveOptions
132         * @see com.digi.xbee.api.models.XBee16BitAddress
133         * @see com.digi.xbee.api.models.XBee64BitAddress 
134         */
135        public IODataSampleRxIndicatorPacket(XBee64BitAddress sourceAddress64, XBee16BitAddress sourceAddress16, int receiveOptions, byte[] rfData) {
136                super(APIFrameType.IO_DATA_SAMPLE_RX_INDICATOR);
137                
138                if (sourceAddress64 == null)
139                        throw new NullPointerException("64-bit source address cannot be null.");
140                if (sourceAddress16 == null)
141                        throw new NullPointerException("16-bit source address cannot be null.");
142                if (receiveOptions < 0 || receiveOptions > 255)
143                        throw new IllegalArgumentException("Receive options value must be between 0 and 255.");
144                
145                this.sourceAddress64 = sourceAddress64;
146                this.sourceAddress16 = sourceAddress16;
147                this.receiveOptions = receiveOptions;
148                this.rfData = rfData;
149                if (rfData != null && rfData.length >= 5)
150                        ioSample = new IOSample(rfData);
151                else
152                        ioSample = null;
153                this.logger = LoggerFactory.getLogger(IODataSampleRxIndicatorPacket.class);
154        }
155        
156        /*
157         * (non-Javadoc)
158         * @see com.digi.xbee.api.packet.XBeeAPIPacket#getAPIData()
159         */
160        @Override
161        protected byte[] getAPIPacketSpecificData() {
162                ByteArrayOutputStream os = new ByteArrayOutputStream();
163                try {
164                        os.write(sourceAddress64.getValue());
165                        os.write(sourceAddress16.getValue());
166                        os.write(receiveOptions);
167                        if (rfData != null)
168                                os.write(rfData);
169                } catch (IOException e) {
170                        logger.error(e.getMessage(), e);
171                }
172                return os.toByteArray();
173        }
174        
175        /*
176         * (non-Javadoc)
177         * @see com.digi.xbee.api.packet.XBeeAPIPacket#needsAPIFrameID()
178         */
179        @Override
180        public boolean needsAPIFrameID() {
181                return false;
182        }
183        
184        /*
185         * (non-Javadoc)
186         * @see com.digi.xbee.api.packet.XBeeAPIPacket#isBroadcast()
187         */
188        @Override
189        public boolean isBroadcast() {
190                return ByteUtils.isBitEnabled(getReceiveOptions(), 1);
191        }
192        
193        /**
194         * Returns the 64-bit sender/source address. 
195         * 
196         * @return The 64-bit sender/source address.
197         * 
198         * @see com.digi.xbee.api.models.XBee64BitAddress
199         */
200        public XBee64BitAddress get64bitSourceAddress() {
201                return sourceAddress64;
202        }
203        
204        /**
205         * Returns the 16-bit sender/source address. 
206         * 
207         * @return 16-bit sender/source address.
208         * 
209         * @see com.digi.xbee.api.models.XBee16BitAddress
210         */
211        public XBee16BitAddress get16bitSourceAddress() {
212                return sourceAddress16;
213        }
214        
215        /**
216         * Returns the receive options. 
217         * 
218         * @return Receive options.
219         * 
220         * @see com.digi.xbee.api.models.XBeeReceiveOptions
221         */
222        public int getReceiveOptions() {
223                return receiveOptions;
224        }
225        
226        /**
227         * Returns the IO sample corresponding to the data contained in the packet.
228         * 
229         * @return The IO sample of the packet, {@code null} if the packet has not 
230         *         any data or if the sample could not be generated correctly.
231         * 
232         * @see com.digi.xbee.api.io.IOSample
233         */
234        public IOSample getIOSample() {
235                return ioSample;
236        }
237        
238        /**
239         * Sets the received RF data.
240         * 
241         * @param rfData Received RF data.
242         */
243        public void setRFData(byte[] rfData) {
244                if (rfData == null)
245                        this.rfData = null;
246                else
247                        this.rfData = Arrays.copyOf(rfData, rfData.length);
248                
249                // Modify the ioSample accordingly.
250                if (rfData != null && rfData.length >= 5)
251                        ioSample = new IOSample(this.rfData);
252                else
253                        ioSample = null;
254        }
255        
256        /**
257         * Returns the received RF data.
258         * 
259         * @return Received RF data.
260         */
261        public byte[] getRFData() {
262                if (rfData == null)
263                        return null;
264                return Arrays.copyOf(rfData, rfData.length);
265        }
266        
267        /*
268         * (non-Javadoc)
269         * @see com.digi.xbee.api.packet.XBeeAPIPacket#getAPIPacketParameters()
270         */
271        @Override
272        public LinkedHashMap<String, String> getAPIPacketParameters() {
273                LinkedHashMap<String, String> parameters = new LinkedHashMap<String, String>();
274                parameters.put("64-bit source address", HexUtils.prettyHexString(sourceAddress64.toString()));
275                parameters.put("16-bit source address", HexUtils.prettyHexString(sourceAddress16.toString()));
276                parameters.put("Receive options", HexUtils.prettyHexString(HexUtils.integerToHexString(receiveOptions, 1)));
277                if (ioSample != null) {
278                        parameters.put("Number of samples", HexUtils.prettyHexString(HexUtils.integerToHexString(1, 1))); // There is always 1 sample.
279                        parameters.put("Digital channel mask", HexUtils.prettyHexString(HexUtils.integerToHexString(ioSample.getDigitalMask(), 2)));
280                        parameters.put("Analog channel mask", HexUtils.prettyHexString(HexUtils.integerToHexString(ioSample.getAnalogMask(), 1)));
281                        for (int i = 0; i < 16; i++) {
282                                if (ioSample.hasDigitalValue(IOLine.getDIO(i)))
283                                        parameters.put(IOLine.getDIO(i).getName() + " digital value", ioSample.getDigitalValue(IOLine.getDIO(i)).getName());
284                        }
285                        for (int i = 0; i < 6; i++) {
286                                if (ioSample.hasAnalogValue(IOLine.getDIO(i)))
287                                        parameters.put(IOLine.getDIO(i).getName() + " analog value", HexUtils.prettyHexString(HexUtils.integerToHexString(ioSample.getAnalogValue(IOLine.getDIO(i)), 2)));
288                        }
289                        if (ioSample.hasPowerSupplyValue())
290                                try {
291                                        parameters.put("Power supply value", HexUtils.prettyHexString(HexUtils.integerToHexString(ioSample.getPowerSupplyValue(), 2)));
292                                } catch (OperationNotSupportedException e) { }
293                } else if (rfData != null)
294                        parameters.put("RF data", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(rfData)));
295                return parameters;
296        }
