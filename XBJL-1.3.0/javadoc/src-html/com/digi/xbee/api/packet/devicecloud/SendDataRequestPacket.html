031/**
032 * This class represents a Send Data Request packet. Packet is built
033 * using the parameters of the constructor or providing a valid API payload.
034 *
035 * <p>This frame type is used to send a file of the given name and type to
036 * Device Cloud.</p>
037 *
038 * <p>If the frame ID is non-zero, a {@link SendDataResponsePacket} will be
039 * received.</p>
040 *
041 * @see SendDataResponsePacket
042 * @see com.digi.xbee.api.packet.XBeeAPIPacket
043 * 
044 * @since 1.2.0
045 */
046public class SendDataRequestPacket extends XBeeAPIPacket {
047
048        // Constants.
049        private static final int MIN_API_PAYLOAD_LENGTH = 6; /* 1 (Frame type) + 1 (Frame ID) +  1 (Path length) +
050                                                                                                                                1 (Content type length) + 1 (Transport) + 1 (Options) */
051
052        private static final String ERROR_PAYLOAD_NULL = "Send Data Request packet payload cannot be null.";
053        private static final String ERROR_INCOMPLETE_PACKET = "Incomplete Send Data Request packet.";
054        private static final String ERROR_NOT_VALID = "Payload is not a Send Data Request packet.";
055        private static final String ERROR_FRAME_ID_ILLEGAL = "Frame ID must be between 0 and 255.";
056        private static final String ERROR_OPTIONS_NULL = "Options cannot be null.";
057
058        // Variables.
059        private String path;
060        private String contentType;
061
062        private final int transport = 0x00; // Must be 0 to indicate TCP.
063
064        private SendDataRequestOptions options;
065
066        private byte[] fileData;
067
068        private Logger logger;
069
070        /**
071         * Creates a new {@code SendDataRequestPacket} object from the given payload.
072         *
073         * @param payload The API frame payload. It must start with the frame type
074         *                corresponding to a Send Data Request packet
075         *                ({@code 0x28}). The byte array must be in
076         *                {@code OperatingMode.API} mode.
077         *
078         * @return Parsed Send Data Request packet.
079         *
080         * @throws IllegalArgumentException if {@code payload[0] != APIFrameType.SEND_DATA_REQUEST.getValue()} or
081         *                                  if {@code payload.length < }{@value #MIN_API_PAYLOAD_LENGTH}.
082         * @throws NullPointerException if {@code payload == null}.
083         */
084        public static SendDataRequestPacket createPacket(byte[] payload) {
085                if (payload == null)
086                        throw new NullPointerException(ERROR_PAYLOAD_NULL);
087
088                if (payload.length < MIN_API_PAYLOAD_LENGTH)
089                        throw new IllegalArgumentException(ERROR_INCOMPLETE_PACKET);
090
091                if ((payload[0] & 0xFF) != APIFrameType.SEND_DATA_REQUEST.getValue())
092                        throw new IllegalArgumentException(ERROR_NOT_VALID);
093
094                // payload[0] is the frame type.
095                int index = 1;
096
097                // Frame ID byte.
098                int frameID = payload[index] & 0xFF;
099                index = index + 1;
100
101                // Path length byte.
102                int pathLength = payload[index] & 0xFF;
103                index = index + 1;
104
105                // Path string.
106                String path = null;
107                if (pathLength > 0) {
108                        path = new String(Arrays.copyOfRange(payload, index, index + pathLength));
109                        index = index + pathLength;
110                }
111
112                // Content type length byte.
113                int contentTypeLength = payload[index] & 0xFF;
114                index = index + 1;
115
116                // Content type string.
117                String contentType = null;
118                if (contentTypeLength > 0) {
119                        contentType = new String(Arrays.copyOfRange(payload, index, index + contentTypeLength));
120                        index = index + contentTypeLength;
121                }
122
123                // Transport byte, reserved.
124                index = index + 1;
125
126                // Target length byte.
127                SendDataRequestOptions options = SendDataRequestOptions.get(payload[index] & 0xFF);
128                index = index + 1;
129
130                // Get data.
131                byte[] data = null;
132                if (index < payload.length)
133                        data = Arrays.copyOfRange(payload, index, payload.length);
134
135                return new SendDataRequestPacket(frameID, path, contentType, options, data);
136        }
137
138        /**
139         * Class constructor. Instantiates a new {@code SendDataRequestPacket}
140         * object with the given parameters.
141         *
142         * @param frameID Frame ID.
143         * @param path Path of the file to upload to Device Cloud.
144         * @param contentType Content type of the file to upload.
145         * @param options Options.
146         * @param fileData Data of the file to upload.
147         *
148         * @throws IllegalArgumentException if {@code frameID < 0} or
149         *                                  if {@code frameID > 255}.
150         * @throws NullPointerException if {@code options == null}.
151         *
152         * @see SendDataRequestOptions
153         */
154        public SendDataRequestPacket(int frameID, String path, String contentType, SendDataRequestOptions options, byte[] fileData) {
155                super(APIFrameType.SEND_DATA_REQUEST);
156
157                if (frameID < 0 || frameID > 255)
158                        throw new IllegalArgumentException(ERROR_FRAME_ID_ILLEGAL);
159                if (options == null)
160                        throw new NullPointerException(ERROR_OPTIONS_NULL);
161
162                this.frameID = frameID;
163                this.path = path;
164                this.contentType = contentType;
165                this.options = options;
166                this.fileData = fileData;
167                this.logger = LoggerFactory.getLogger(SendDataRequestPacket.class);
168        }
169
170        /*
171         * (non-Javadoc)
172         * @see com.digi.xbee.api.packet.XBeeAPIPacket#getAPIPacketSpecificData()
173         */
174        @Override
175        protected byte[] getAPIPacketSpecificData() {
176                ByteArrayOutputStream os = new ByteArrayOutputStream();
177                try {
178                        if (path == null)
179                                os.write(0x00);
180                        else {
181                                os.write(path.length());
182                                os.write(path.getBytes());
183                        }
184                        if (contentType == null)
185                                os.write(0x00);
186                        else {
187                                os.write(contentType.length());
188                                os.write(contentType.getBytes());
189                        }
190                        os.write(transport);
191                        os.write(options.getID());
192                        if (fileData != null)
193                                os.write(fileData);
194                } catch (IOException e) {
195                        logger.error(e.getMessage(), e);
196                }
197                return os.toByteArray();
198        }

322        /*
323         * (non-Javadoc)
324         * @see com.digi.xbee.packet.XBeeAPIPacket#getAPIPacketParameters()
325         */
326        @Override
327        public LinkedHashMap<String, String> getAPIPacketParameters() {
328                LinkedHashMap<String, String> parameters = new LinkedHashMap<String, String>();
329                if (path != null) {
330                        parameters.put("Path length", HexUtils.prettyHexString(HexUtils.integerToHexString(path.length(), 1)) + " (" + path.length() + ")");
331                        parameters.put("Path", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(path.getBytes())) + " (" + path + ")");
332                } else
333                        parameters.put("Path length", HexUtils.prettyHexString(HexUtils.integerToHexString(0x00, 1)) + " (0)");
334                if (contentType != null) {
335                        parameters.put("Content Type length", HexUtils.prettyHexString(HexUtils.integerToHexString(contentType.length(), 1)) + " (" + contentType.length() + ")");
336                        parameters.put("Content Type", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(contentType.getBytes())) + " (" + contentType + ")");
337                } else
338                        parameters.put("Content Type length", HexUtils.prettyHexString(HexUtils.integerToHexString(0x00, 1)) + " (0)");
339                parameters.put("Transport", HexUtils.prettyHexString(HexUtils.integerToHexString(transport, 1)));
340                parameters.put("Options", HexUtils.prettyHexString(HexUtils.integerToHexString(options.getID(), 1)) + " (" + options.getName() + ")");
341                if (fileData != null)
342                        parameters.put("Data", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(fileData)));
343                return parameters;
344        }
