034/**
035 * This class represents a TX (Transmit) IPv4 packet. Packet is built
036 * using the parameters of the constructor or providing a valid API payload.
037 *
038 * @see RXIPv4Packet
039 * @see com.digi.xbee.api.packet.XBeeAPIPacket
040 * 
041 * @since 1.2.0
042 */
043public class TXIPv4Packet extends XBeeAPIPacket {
044
045        // Constants.
046        /** This option will close the socket after the transmission. */
047        public static final int OPTIONS_CLOSE_SOCKET = 2;
048        /** This option will leave socket open after the transmission. */
049        public static final int OPTIONS_LEAVE_SOCKET_OPEN = 0;
050
051        private static final int MIN_API_PAYLOAD_LENGTH = 12; /* 1 (Frame type) + 1 (frame ID) + 4 (dest address) + 2 (dest port) +
052                                                                                                                                2 (source port) + 1 (protocol) + 1 (transmit options) */
053
054        private static final String ERROR_PAYLOAD_NULL = "TX IPv4 packet payload cannot be null.";
055        private static final String ERROR_INCOMPLETE_PACKET = "Incomplete TX IPv4 packet.";
056        private static final String ERROR_NOT_TXIPV4 = "Payload is not a TX IPv4 packet.";
057        private static final String ERROR_DEST_ADDR_NULL = "Destination address cannot be null.";
058        private static final String ERROR_PROTOCOL_NULL = "Protocol cannot be null.";
059        private static final String ERROR_FRAME_ID_ILLEGAL = "Frame ID must be between 0 and 255.";
060        private static final String ERROR_PORT_ILLEGAL = "Port must be between 0 and 65535.";
061        private static final String ERROR_OPTIONS_INVALID = "Transmit options can only be " + OPTIONS_CLOSE_SOCKET +
062                        " or " + OPTIONS_LEAVE_SOCKET_OPEN + ".";
063
064        // Variables.
065        private Inet4Address destAddress;
066
067        private int destPort;
068        private int sourcePort;
069        private int transmitOptions;
070
071        private IPProtocol protocol;
072
073        private byte[] data;
074
075        private Logger logger;
076
077        /**
078         * Creates a new {@code TXIPv4Packet} object from the given payload.
079         *
080         * @param payload The API frame payload. It must start with the frame type
081         *                corresponding to a TX IPv4 packet ({@code 0x20}).
082         *                The byte array must be in {@code OperatingMode.API} mode.
083         *
084         * @return Parsed TX IPv4 packet.
085         *
086         * @throws IllegalArgumentException if {@code payload[0] != APIFrameType.TX_IPV4.getValue()} or
087         *                                  if {@code payload.length < }{@value #MIN_API_PAYLOAD_LENGTH}.
088         * @throws NullPointerException if {@code payload == null}.
089         */
090        public static TXIPv4Packet createPacket(byte[] payload) {
091                if (payload == null)
092                        throw new NullPointerException(ERROR_PAYLOAD_NULL);
093
094                if (payload.length < MIN_API_PAYLOAD_LENGTH)
095                        throw new IllegalArgumentException(ERROR_INCOMPLETE_PACKET);
096
097                if ((payload[0] & 0xFF) != APIFrameType.TX_IPV4.getValue())
098                        throw new IllegalArgumentException(ERROR_NOT_TXIPV4);
099
100                // payload[0] is the frame type.
101                int index = 1;
102
103                // Frame ID byte.
104                int frameID = payload[index] & 0xFF;
105                index = index + 1;
106
107                // 4 bytes of IP destination address.
108                Inet4Address destAddress;
109                try {
110                        destAddress = (Inet4Address) Inet4Address.getByAddress(Arrays.copyOfRange(payload, index, index + 4));
111                } catch (UnknownHostException e) {
112                        throw new IllegalArgumentException(e);
113                }
114                index = index + 4;
115
116                // 2 bytes of destination port.
117                int destPort = (payload[index] & 0xFF) << 8 | payload[index + 1] & 0xFF;
118                index = index + 2;
119
120                // 2 bytes of source port.
121                int sourcePort = (payload[index] & 0xFF) << 8 | payload[index + 1] & 0xFF;
122                index = index + 2;
123
124                // Protocol byte.
125                IPProtocol protocol = IPProtocol.get(payload[index] & 0xFF);
126                index = index + 1;
127
128                // Transmit options byte.
129                int transmitOptions = payload[index] & 0xFF;
130                index = index + 1;
131
132                // Get data.
133                byte[] data = null;
134                if (index < payload.length)
135                        data = Arrays.copyOfRange(payload, index, payload.length);
136
137                return new TXIPv4Packet(frameID, destAddress, destPort, sourcePort, protocol, transmitOptions, data);
138        }

140        /**
141         * Class constructor. Instantiates a new {@code TXIPv4Packet} object with
142         * the given parameters.
143         *
144         * @param frameID Frame ID.
145         * @param destAddress IP address of the destination device.
146         * @param destPort Destination port number.
147         * @param sourcePort Source port number.
148         * @param protocol Protocol used for transmitted data.
149         * @param transmitOptions Transmit options bitfield. Can be
150         *                        {@link #OPTIONS_LEAVE_SOCKET_OPEN} or
151         *                        {@link #OPTIONS_CLOSE_SOCKET}.
152         * @param data Transmit data bytes.
153         *
154         * @throws IllegalArgumentException if {@code frameID < 0} or
155         *                                  if {@code frameID > 255} or
156         *                                  if {@code destPort < 0} or
157         *                                  if {@code destPort > 65535} or
158         *                                  if {@code sourcePort < 0} or
159         *                                  if {@code sourcePort > 65535} or
160         *                                  if {@code transmitOptions} are invalid.
161         * @throws NullPointerException if {@code destAddress == null} or
162         *                              if {@code protocol == null}.
163         *
164         * @see IPProtocol
165         * @see java.net.Inet4Address
166         */
167        public TXIPv4Packet(int frameID, Inet4Address destAddress, int destPort, int sourcePort,
168                        IPProtocol protocol, int transmitOptions, byte[] data) {
169                super(APIFrameType.TX_IPV4);
170
171                if (frameID < 0 || frameID > 255)
172                        throw new IllegalArgumentException(ERROR_FRAME_ID_ILLEGAL);
173                if (destPort < 0 || destPort > 65535)
174                        throw new IllegalArgumentException(ERROR_PORT_ILLEGAL);
175                if (sourcePort < 0 || sourcePort > 65535)
176                        throw new IllegalArgumentException(ERROR_PORT_ILLEGAL);
177                if (transmitOptions != OPTIONS_CLOSE_SOCKET && transmitOptions != OPTIONS_LEAVE_SOCKET_OPEN)
178                        throw new IllegalArgumentException(ERROR_OPTIONS_INVALID);
179                if (destAddress == null)
180                        throw new NullPointerException(ERROR_DEST_ADDR_NULL);
181                if (protocol == null)
182                        throw new NullPointerException(ERROR_PROTOCOL_NULL);
183
184                this.frameID = frameID;
185                this.destAddress = destAddress;
186                this.destPort = destPort;
187                this.sourcePort = sourcePort;
188                this.protocol = protocol;
189                this.transmitOptions = transmitOptions;
190                this.data = data;
191                this.logger = LoggerFactory.getLogger(TXIPv4Packet.class);
192        }

194        /*
195         * (non-Javadoc)
196         * @see com.digi.xbee.api.packet.XBeeAPIPacket#getAPIPacketSpecificData()
197         */
198        @Override
199        protected byte[] getAPIPacketSpecificData() {
200                ByteArrayOutputStream os = new ByteArrayOutputStream();
201                try {
202                        os.write(destAddress.getAddress());
203                        os.write(destPort >> 8);
204                        os.write(destPort);
205                        os.write(sourcePort >> 8);
206                        os.write(sourcePort);
207                        os.write(protocol.getID());
208                        os.write(transmitOptions);
209                        if (data != null)
210                                os.write(data);
211                } catch (IOException e) {
212                        logger.error(e.getMessage(), e);
213                }
214                return os.toByteArray();
215        }

409        /*
410         * (non-Javadoc)
411         * @see com.digi.xbee.packet.XBeeAPIPacket#getAPIPacketParameters()
412         */
413        @Override
414        public LinkedHashMap<String, String> getAPIPacketParameters() {
415                LinkedHashMap<String, String> parameters = new LinkedHashMap<String, String>();
416                parameters.put("Destination address", HexUtils.prettyHexString(destAddress.getAddress()) + " (" + destAddress.getHostAddress() + ")");
417                parameters.put("Destination port", HexUtils.prettyHexString(HexUtils.integerToHexString(destPort, 2)) + " (" + destPort + ")");
418                parameters.put("Source port", HexUtils.prettyHexString(HexUtils.integerToHexString(sourcePort, 2)) + " (" + sourcePort + ")");
419                parameters.put("Protocol", HexUtils.prettyHexString(HexUtils.integerToHexString(protocol.getID(), 1)) + " (" + protocol.getName() + ")");
420                parameters.put("Transmit options", HexUtils.prettyHexString(HexUtils.integerToHexString(transmitOptions, 1)));
421                if (data != null)
422                        parameters.put("Data", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(data)));
423                return parameters;
424        }
