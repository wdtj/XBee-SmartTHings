033/**
034 * This class represents a TX (Transmit) request with TLS profile packet.
035 * The packet implies the use of TLS and behaves similar to the TX Request
036 * (0x20) frame, with the protocol field replaced with a TLS Profile field to
037 * choose from the profiles configured with the $0, $1, and $2 configuration 
038 * commands. Packet is built using the parameters of the constructor or
039 * providing a valid API payload.
040 *
041 * @see TXIPv4Packet
042 * @see com.digi.xbee.api.packet.XBeeAPIPacket
043 * 
044 * @since 1.3.0
045 */
046public class TXTLSProfilePacket extends XBeeAPIPacket {
047
048        // Constants.
049        /** This option will close the socket after the transmission. */
050        public static final int OPTIONS_CLOSE_SOCKET = TXIPv4Packet.OPTIONS_CLOSE_SOCKET;
051        /** This option will leave socket open after the transmission. */
052        public static final int OPTIONS_LEAVE_SOCKET_OPEN = TXIPv4Packet.OPTIONS_LEAVE_SOCKET_OPEN;
053
054        private static final int MIN_API_PAYLOAD_LENGTH = 12; /* 1 (Frame type) + 1 (frame ID) + 4 (dest address) + 2 (dest port) +
055                                                                                                                                2 (source port) + 1 (profile) + 1 (transmit options) */
056
057        private static final String ERROR_PAYLOAD_NULL = "TX request with TLS profile packet payload cannot be null.";
058        private static final String ERROR_INCOMPLETE_PACKET = "Incomplete TX request with TLS profile packet.";
059        private static final String ERROR_NOT_TXTLS = "Payload is not a TX request with TLS profile packet.";
060        private static final String ERROR_DEST_ADDR_NULL = "Destination address cannot be null.";
061        private static final String ERROR_FRAME_ID_ILLEGAL = "Frame ID must be between 0 and 255.";
062        private static final String ERROR_PORT_ILLEGAL = "Port must be between 0 and 65535.";
063        private static final String ERROR_PROFILE_ILLEGAL = "Profile must be between 0 and 255.";
064        private static final String ERROR_OPTIONS_INVALID = "Transmit options can only be " + OPTIONS_CLOSE_SOCKET +
065                        " or " + OPTIONS_LEAVE_SOCKET_OPEN + ".";
066
067        // Variables.
068        private Inet4Address destAddress;
069
070        private int destPort;
071        private int sourcePort;
072        private int transmitOptions;
073        private int profile;
074
075        private byte[] data;
076
077        private Logger logger;
078
079        /**
080         * Creates a new {@code TXTLSProfilePacket} object from the given payload.
081         *
082         * @param payload The API frame payload. It must start with the frame type
083         *                corresponding to a TX request with TLS profile packet
084         *                ({@code 0x23}). The byte array must be in
085         *                {@code OperatingMode.API} mode.
086         *
087         * @return Parsed TX request with TLS profile packet.
088         *
089         * @throws IllegalArgumentException if {@code payload[0] != APIFrameType.TX_REQUEST_TLS_PROFILE.getValue()} or
090         *                                  if {@code payload.length < }{@value #MIN_API_PAYLOAD_LENGTH}.
091         * @throws NullPointerException if {@code payload == null}.
092         */
093        public static TXTLSProfilePacket createPacket(byte[] payload) {
094                if (payload == null)
095                        throw new NullPointerException(ERROR_PAYLOAD_NULL);
096
097                if (payload.length < MIN_API_PAYLOAD_LENGTH)
098                        throw new IllegalArgumentException(ERROR_INCOMPLETE_PACKET);
099
100                if ((payload[0] & 0xFF) != APIFrameType.TX_REQUEST_TLS_PROFILE.getValue())
101                        throw new IllegalArgumentException(ERROR_NOT_TXTLS);
102
103                // payload[0] is the frame type.
104                int index = 1;
105
106                // Frame ID byte.
107                int frameID = payload[index] & 0xFF;
108                index = index + 1;
109
110                // 4 bytes of IP destination address.
111                Inet4Address destAddress;
112                try {
113                        destAddress = (Inet4Address) Inet4Address.getByAddress(Arrays.copyOfRange(payload, index, index + 4));
114                } catch (UnknownHostException e) {
115                        throw new IllegalArgumentException(e);
116                }
117                index = index + 4;
118
119                // 2 bytes of destination port.
120                int destPort = (payload[index] & 0xFF) << 8 | payload[index + 1] & 0xFF;
121                index = index + 2;
122
123                // 2 bytes of source port.
124                int sourcePort = (payload[index] & 0xFF) << 8 | payload[index + 1] & 0xFF;
125                index = index + 2;
126
127                // Profile byte.
128                int profile = payload[index] & 0xFF;
129                index = index + 1;
130
131                // Transmit options byte.
132                int transmitOptions = payload[index] & 0xFF;
133                index = index + 1;
134
135                // Get data.
136                byte[] data = null;
137                if (index < payload.length)
138                        data = Arrays.copyOfRange(payload, index, payload.length);
139
140                return new TXTLSProfilePacket(frameID, destAddress, destPort, sourcePort, profile, transmitOptions, data);
141        }

143        /**
144         * Class constructor. Instantiates a new {@code TXTLSProfilePacket} object
145         * with the given parameters.
146         *
147         * @param frameID Frame ID.
148         * @param destAddress IP address of the destination device.
149         * @param destPort Destination port number.
150         * @param sourcePort Source port number.
151         * @param profile Zero-indexed number that indicates the profile as
152         *                specified by the corresponding $num command.
153         * @param transmitOptions Transmit options bitfield. Can be
154         *                        {@link #OPTIONS_LEAVE_SOCKET_OPEN} or
155         *                        {@link #OPTIONS_CLOSE_SOCKET}.
156         * @param data Transmit data bytes.
157         *
158         * @throws IllegalArgumentException if {@code frameID < 0} or
159         *                                  if {@code frameID > 255} or
160         *                                  if {@code destPort < 0} or
161         *                                  if {@code destPort > 65535} or
162         *                                  if {@code sourcePort < 0} or
163         *                                  if {@code sourcePort > 65535} or
164         *                                  if {@code profile < 0} or
165         *                                  if {@code profile > 255} or
166         *                                  if {@code transmitOptions} are invalid.
167         * @throws NullPointerException if {@code destAddress == null}.
168         *
169         * @see java.net.Inet4Address
170         */
171        public TXTLSProfilePacket(int frameID, Inet4Address destAddress, int destPort, int sourcePort,
172                        int profile, int transmitOptions, byte[] data) {
173                super(APIFrameType.TX_REQUEST_TLS_PROFILE);
174
175                if (frameID < 0 || frameID > 255)
176                        throw new IllegalArgumentException(ERROR_FRAME_ID_ILLEGAL);
177                if (destPort < 0 || destPort > 65535)
178                        throw new IllegalArgumentException(ERROR_PORT_ILLEGAL);
179                if (sourcePort < 0 || sourcePort > 65535)
180                        throw new IllegalArgumentException(ERROR_PORT_ILLEGAL);
181                if (transmitOptions != OPTIONS_CLOSE_SOCKET && transmitOptions != OPTIONS_LEAVE_SOCKET_OPEN)
182                        throw new IllegalArgumentException(ERROR_OPTIONS_INVALID);
183                if (profile < 0 || profile > 255)
184                        throw new IllegalArgumentException(ERROR_PROFILE_ILLEGAL);
185                if (destAddress == null)
186                        throw new NullPointerException(ERROR_DEST_ADDR_NULL);
187
188                this.frameID = frameID;
189                this.destAddress = destAddress;
190                this.destPort = destPort;
191                this.sourcePort = sourcePort;
192                this.profile = profile;
193                this.transmitOptions = transmitOptions;
194                this.data = data;
195                this.logger = LoggerFactory.getLogger(TXTLSProfilePacket.class);
196        }

198        /*
199         * (non-Javadoc)
200         * @see com.digi.xbee.api.packet.XBeeAPIPacket#getAPIPacketSpecificData()
201         */
202        @Override
203        protected byte[] getAPIPacketSpecificData() {
204                ByteArrayOutputStream os = new ByteArrayOutputStream();
205                try {
206                        os.write(destAddress.getAddress());
207                        os.write(destPort >> 8);
208                        os.write(destPort);
209                        os.write(sourcePort >> 8);
210                        os.write(sourcePort);
211                        os.write(profile);
212                        os.write(transmitOptions);
213                        if (data != null)
214                                os.write(data);
215                } catch (IOException e) {
216                        logger.error(e.getMessage(), e);
217                }
218                return os.toByteArray();
219        }

411        /*
412         * (non-Javadoc)
413         * @see com.digi.xbee.packet.XBeeAPIPacket#getAPIPacketParameters()
414         */
415        @Override
416        public LinkedHashMap<String, String> getAPIPacketParameters() {
417                LinkedHashMap<String, String> parameters = new LinkedHashMap<String, String>();
418                parameters.put("Destination address", HexUtils.prettyHexString(destAddress.getAddress()) + " (" + destAddress.getHostAddress() + ")");
419                parameters.put("Destination port", HexUtils.prettyHexString(HexUtils.integerToHexString(destPort, 2)) + " (" + destPort + ")");
420                parameters.put("Source port", HexUtils.prettyHexString(HexUtils.integerToHexString(sourcePort, 2)) + " (" + sourcePort + ")");
421                parameters.put("Profile", HexUtils.prettyHexString(HexUtils.integerToHexString(profile, 1)) + " (" + profile + ")");
422                parameters.put("Transmit options", HexUtils.prettyHexString(HexUtils.integerToHexString(transmitOptions, 1)));
423                if (data != null)
424                        parameters.put("Data", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(data)));
425                return parameters;
426        }
