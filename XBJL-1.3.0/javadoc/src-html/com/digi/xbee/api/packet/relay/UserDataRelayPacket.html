016package com.digi.xbee.api.packet.relay;
017
018import java.io.ByteArrayOutputStream;
019import java.io.IOException;
020import java.util.Arrays;
021import java.util.LinkedHashMap;
022
023import com.digi.xbee.api.models.XBeeLocalInterface;
024import com.digi.xbee.api.packet.APIFrameType;
025import com.digi.xbee.api.packet.XBeeAPIPacket;
026import com.digi.xbee.api.utils.HexUtils;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030/**
031 * This class represents a User Data Relay packet. Packet is built using the
032 * parameters of the constructor.
033 *
034 * The User Data Relay packet allows for data to come in on an interface with
035 * a designation of the target interface for the data to be output on.
036 *
037 * The destination interface must be one of the interfaces found in the
038 * corresponding enumerator. See {@link XBeeLocalInterface}.
039 *
040 * @see UserDataRelayOutputPacket
041 * @see com.digi.xbee.api.packet.XBeeAPIPacket
042 *
043 * @since 1.3.0
044 */
045public class UserDataRelayPacket extends XBeeAPIPacket {
046
047        // Constants.
048        private static final int MIN_API_PAYLOAD_LENGTH = 3; /* 1 (Frame type) + 1 (frame ID) + 1 (dest interface) */
049
050        private static final String ERROR_PAYLOAD_NULL = "User Data Relay packet payload cannot be null.";
051        private static final String ERROR_INCOMPLETE_PACKET = "Incomplete User Data Relay packet.";
052        private static final String ERROR_NOT_USER_DATA_RELAY = "Payload is not a User Data Relay packet.";
053        private static final String ERROR_INTERFACE_NULL = "Destination interface cannot be null.";
054        private static final String ERROR_FRAME_ID_ILLEGAL = "Frame ID must be between 0 and 255.";
055
056        // Variables.
057        private XBeeLocalInterface localInterface;
058
059        private byte[] data;
060
061        private Logger logger;
062
063        /**
064         * Creates a new {@code UserDataRelayPacket} object from the given payload.
065         *
066         * @param payload The API frame payload. It must start with the frame type
067         *                corresponding to a User Data Relay packet ({@code 0x2D}).
068         *                The byte array must be in {@code OperatingMode.API} mode.
069         *
070         * @return Parsed User Data Relay packet.
071         *
072         * @throws IllegalArgumentException if {@code payload[0] != APIFrameType.USER_DATA_RELAY.getValue()} or
073         *                                  if {@code payload.length < }{@value #MIN_API_PAYLOAD_LENGTH}.
074         * @throws NullPointerException if {@code payload == null}.
075         */
076        public static UserDataRelayPacket createPacket(byte[] payload) {
077                if (payload == null)
078                        throw new NullPointerException(ERROR_PAYLOAD_NULL);
079
080                if (payload.length < MIN_API_PAYLOAD_LENGTH)
081                        throw new IllegalArgumentException(ERROR_INCOMPLETE_PACKET);
082
083                if ((payload[0] & 0xFF) != APIFrameType.USER_DATA_RELAY.getValue())
084                        throw new IllegalArgumentException(ERROR_NOT_USER_DATA_RELAY);
085
086                // payload[0] is the frame type.
087                int index = 1;
088
089                // Frame ID byte.
090                int frameID = payload[index] & 0xFF;
091                index = index + 1;
092
093                // Destination interface byte.
094                XBeeLocalInterface destInterface = XBeeLocalInterface.get(payload[index] & 0xFF);
095                index = index + 1;
096
097                // Get data.
098                byte[] data = null;
099                if (index < payload.length)
100                        data = Arrays.copyOfRange(payload, index, payload.length);
101
102                return new UserDataRelayPacket(frameID, destInterface, data);
103        }
104
105        /**
106         * Class constructor. Instantiates a new {@code UserDataRelayPacket}
107         * object with the given parameters.
108         *
109         * @param frameID Frame ID.
110         * @param localInterface The destination {@code XBeeLocalInterface}.
111         * @param data RF Data that is sent to the destination interface.
112         *
113         * @throws IllegalArgumentException if {@code frameID < 0} or
114         *                                  if {@code frameID > 255}.
115         * @throws NullPointerException if {@code localInterface == null}.
116         */
117        public UserDataRelayPacket(int frameID, XBeeLocalInterface localInterface, byte[] data) {
118                super(APIFrameType.USER_DATA_RELAY);
119
120                if (frameID < 0 || frameID > 255)
121                        throw new IllegalArgumentException(ERROR_FRAME_ID_ILLEGAL);
122                if (localInterface == null)
123                        throw new NullPointerException(ERROR_INTERFACE_NULL);
124
125                this.frameID = frameID;
126                this.localInterface = localInterface;
127                this.data = data;
128                this.logger = LoggerFactory.getLogger(UserDataRelayPacket.class);
129        }
130
131        @Override
132        public byte[] getAPIPacketSpecificData() {
133                ByteArrayOutputStream os = new ByteArrayOutputStream();
134                try {
135                        os.write(localInterface.getID());
136                        if (data != null)
137                                os.write(data);
138                } catch (IOException e) {
139                        logger.error(e.getMessage(), e);
140                }
141                return os.toByteArray();
142        }
143
144        @Override
145        public boolean needsAPIFrameID() {
146                return true;
147        }
148
149        @Override
150        public boolean isBroadcast() {
151                return false;
152        }
153        
154        /**
155         * Sets the destination XBee local interface.
156         *
157         * @param localInterface The new destination interface.
158         *
159         * @throws NullPointerException if {@code localInterface == null}.
160         *
161         * @see #getDestinationInterface()
162         * @see XBeeLocalInterface
163         */
164        public void setDestinationInterface(XBeeLocalInterface localInterface) {
165                if (localInterface == null)
166                        throw new NullPointerException(ERROR_INTERFACE_NULL);
167                this.localInterface = localInterface;
168        }
169        
170        /**
171         * Retrieves the the destination XBee local interface.
172         *
173         * @return The the destination interface.
174         *
175         * @see #setDestinationInterface(XBeeLocalInterface)
176         * @see XBeeLocalInterface
177         */
178        public XBeeLocalInterface getDestinationInterface() {
179                return localInterface;
180        }
181        
182        /**
183         * Sets the data to send.
184         *
185         * @param data Data to send.
186         *
187         * @see #getData()
188         */
189        public void setData(byte[] data) {
190                this.data = data;
191        }
192        
193        /**
194         * Retrieves the data to send.
195         *
196         * @return Data to send.
197         *
198         * @see #setData(byte[])
199         */
200        public byte[] getData() {
201                return data;
202        }
203
204        @Override
205        public LinkedHashMap<String, String> getAPIPacketParameters() {
206                LinkedHashMap<String, String> parameters = new LinkedHashMap<>();
207                parameters.put("Destination interface", String.format("%s (%s)", HexUtils.prettyHexString(HexUtils.integerToHexString(
208                                localInterface.getID(), 1)), localInterface.getDescription()));
209                if (data != null)
210                        parameters.put("Data", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(data)));
211                return parameters;
212        }
213}
