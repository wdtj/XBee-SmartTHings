035/**
036 * This class represents a CoAP Rx Response packet. Packet is built using the 
037 * parameters of the constructor.
038 * 
039 * <p>When the module receives a CoAP packet, it is sent out the UART using 
040 * this message type.</p>
041 * 
042 * <p>This packet is received when external devices send CoAP Tx Request 
043 * packets to the module. See 
044 * {@link com.digi.xbee.api.packet.thread.CoAPTxRequestPacket}.</p>
045 * 
046 * @see CoAPTxRequestPacket
047 * @see com.digi.xbee.api.packet.XBeeAPIPacket
048 * 
049 * @since 1.2.1
050 */
051public class CoAPRxResponsePacket extends XBeeAPIPacket {
052
053        // Constants.
054        private static final int MIN_API_PAYLOAD_LENGTH = 41; /* 1 (Frame type) + 1 (frame ID) + 16 (dest address) + 16 (source address) +
055                                                                                                                                2 (dest port) + 2 (source port) + 1 (protocol) + 2 (Restful Status) */
056
057        private static final String ERROR_PAYLOAD_NULL = "CoAP Rx Response packet payload cannot be null.";
058        private static final String ERROR_INCOMPLETE_PACKET = "Incomplete CoAP Rx Response packet.";
059        private static final String ERROR_NOT_COAP_RX_RESP = "Payload is not a CoAP Rx Response packet.";
060        private static final String ERROR_DEST_ADDR_NULL = "Destination address cannot be null.";
061        private static final String ERROR_SOURCE_ADDR_NULL = "Source address cannot be null.";
062        private static final String ERROR_PROTOCOL_NULL = "Protocol cannot be null.";
063        private static final String ERROR_STATUS_NULL = "RESTFul status cannot be null.";
064        private static final String ERROR_FRAME_ID_ILLEGAL = "Frame ID must be between 0 and 255.";
065        private static final String ERROR_DEST_PORT_ILLEGAL = "Destination port must be between 0 and 65535.";
066        private static final String ERROR_SOURCE_PORT_ILLEGAL = "Source port must be between 0 and 65535.";
067
068        private static final String OPERATION_EXCEPTION = "Operation not supported in this module.";
069
070        // Variables.
071        private Inet6Address destAddress;
072        private Inet6Address sourceAddress;
073

075        private int sourcePort;

085        /**
086         * Creates a new {@code CoAPRxResponsePacket} object from the given payload.
087         *
088         * @param payload The API frame payload. It must start with the frame type
089         *                corresponding to a CoAP Rx Response packet ({@code 0x9C}).
090         *                The byte array must be in {@code OperatingMode.API} mode.
091         *
092         * @return Parsed CoAP Tx Request packet.
093         *
094         * @throws IllegalArgumentException if {@code payload[0] != APIFrameType.COAP_RX_RESPONSE.getValue()} or
095         *                                  if {@code payload.length < }{@value #MIN_API_PAYLOAD_LENGTH}.
096         * @throws NullPointerException if {@code payload == null}.
097         */
098        public static CoAPRxResponsePacket createPacket(byte[] payload) {
099                if (payload == null)
100                        throw new NullPointerException(ERROR_PAYLOAD_NULL);
101
102                if (payload.length < MIN_API_PAYLOAD_LENGTH)
103                        throw new IllegalArgumentException(ERROR_INCOMPLETE_PACKET);
104
105                if ((payload[0] & 0xFF) != APIFrameType.COAP_RX_RESPONSE.getValue())
106                        throw new IllegalArgumentException(ERROR_NOT_COAP_RX_RESP);
107
108                // payload[0] is the frame type.
109                int index = 1;
110
111                // Frame ID byte.
112                int frameID = payload[index] & 0xFF;
113                index = index + 1;
114
115                // 16 bytes of IP destination address.
116                Inet6Address destAddress;
117                try {
118                        destAddress = (Inet6Address) Inet6Address.getByAddress(Arrays.copyOfRange(payload, index, index + 16));
119                } catch (UnknownHostException e) {
120                        throw new IllegalArgumentException(e);
121                }
122                index = index + 16;
123
124                // 16 bytes of IP source address.
125                Inet6Address sourceAddress;
126                try {
127                        sourceAddress = (Inet6Address) Inet6Address.getByAddress(Arrays.copyOfRange(payload, index, index + 16));
128                } catch (UnknownHostException e) {
129                        throw new IllegalArgumentException(e);
130                }
131                index = index + 16;
132
133                // 2 bytes of destination port.
134                int destPort = (payload[index] & 0xFF) << 8 | payload[index + 1] & 0xFF;
135                index = index + 2;
136
137                // 2 bytes of source port.
138                int sourcePort = (payload[index] & 0xFF) << 8 | payload[index + 1] & 0xFF;
139                index = index + 2;
140
141                // Transmit options.
142                IPProtocol protocol = IPProtocol.get(payload[index] & 0xFF);
143                index = index + 1;
144
145                // 2 bytes of RESTFul status.
146                RestFulStatusEnum restFulStatus = RestFulStatusEnum.get((payload[index] & 0xFF) << 8 | payload[index + 1] & 0xFF);
147                index = index + 2;
148
149                // Get data.
150                byte[] rfData = null;
151                if (index < payload.length)
152                        rfData = Arrays.copyOfRange(payload, index, payload.length);
153
154                return new CoAPRxResponsePacket(frameID, destAddress, sourceAddress, destPort, sourcePort, protocol, restFulStatus, rfData);
155        }

157        /**
158         * Class constructor. Instantiates a new {@code CoAPRxResponsePacket} object with
159         * the given parameters.
160         *
161         * @param frameID Frame ID.
162         * @param destAddress IPv6 address of the destination device.
163         * @param sourceAddress IPv6 address of the source device.
164         * @param destPort Destination port number.
165         * @param sourcePort Source port number.
166         * @param protocol Protocol used for transmitted data.
167         * @param restFulStatus RESTful status.
168         * @param rfData RF transmit data bytes.
169         *
170         * @throws IllegalArgumentException if {@code frameID < 0} or
171         *                                  if {@code frameID > 255} or
172         *                                  if {@code destPort < 0} or
173         *                                  if {@code destPort > 65535} or
174         *                                  if {@code sourcePort < 0} or
175         *                                  if {@code sourcePort > 65535}.
176         * @throws NullPointerException if {@code destAddress == null} or 
177         *                              if {@code sourceAddress == null} or
178         *                              if {@code protocol == null} or
179         *                              if {@code restFulStatus == null}.
180         *
181         * @see java.net.Inet6Address
182         * @see com.digi.xbee.api.models.IPProtocol
183         * @see com.digi.xbee.api.models.RestFulStatusEnum
184         */
185        public CoAPRxResponsePacket(int frameID, Inet6Address destAddress, Inet6Address sourceAddress,
186                        int destPort, int sourcePort, IPProtocol protocol, RestFulStatusEnum restFulStatus, byte[] rfData) {
187                super(APIFrameType.COAP_RX_RESPONSE);
188
189                if (frameID < 0 || frameID > 255)
190                        throw new IllegalArgumentException(ERROR_FRAME_ID_ILLEGAL);
191                if (destAddress == null)
192                        throw new NullPointerException(ERROR_DEST_ADDR_NULL);
193                if (sourceAddress == null)
194                        throw new NullPointerException(ERROR_SOURCE_ADDR_NULL);
195                if (destPort < 0 || destPort > 65535)
196                        throw new IllegalArgumentException(ERROR_DEST_PORT_ILLEGAL);
197                if (sourcePort < 0 || sourcePort > 65535)
198                        throw new IllegalArgumentException(ERROR_SOURCE_PORT_ILLEGAL);
199                if (protocol == null)
200                        throw new NullPointerException(ERROR_PROTOCOL_NULL);
201                if (restFulStatus == null)
202                        throw new NullPointerException(ERROR_STATUS_NULL);
203
204                this.frameID = frameID;
205                this.destAddress = destAddress;
206                this.sourceAddress = sourceAddress;
207                this.destPort = destPort;
208                this.sourcePort = sourcePort;
209                this.protocol = protocol;
210                this.restFulStatus = restFulStatus;
211                this.data = rfData;
212                this.logger = LoggerFactory.getLogger(CoAPRxResponsePacket.class);
213        }

215        /*
216         * (non-Javadoc)
217         * @see com.digi.xbee.api.packet.XBeeAPIPacket#getAPIPacketSpecificData()
218         */
219        @Override
220        protected byte[] getAPIPacketSpecificData() {
221                ByteArrayOutputStream os = new ByteArrayOutputStream();
222                try {
223                        os.write(destAddress.getAddress());
224                        os.write(sourceAddress.getAddress());
225                        os.write(ByteUtils.shortToByteArray((short)destPort));
226                        os.write(ByteUtils.shortToByteArray((short)sourcePort));
227                        os.write(protocol.getID());
228                        os.write(ByteUtils.shortToByteArray((short)restFulStatus.getID()));
229                        if (data != null)
230                                os.write(data);
231                } catch (IOException e) {
232                        logger.error(e.getMessage(), e);
233                }
234                return os.toByteArray();
235        }

452        /*
453         * (non-Javadoc)
454         * @see com.digi.xbee.packet.XBeeAPIPacket#getAPIPacketParameters()
455         */
456        @Override
457        public LinkedHashMap<String, String> getAPIPacketParameters() {
458                LinkedHashMap<String, String> parameters = new LinkedHashMap<String, String>();
459                parameters.put("Destination address", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(destAddress.getAddress())) +
460                                " (" + destAddress.getHostAddress() + ")");
461                parameters.put("Source address", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(sourceAddress.getAddress())) +
462                                " (" + sourceAddress.getHostAddress() + ")");
463                parameters.put("Destination port", HexUtils.prettyHexString(HexUtils.integerToHexString(destPort, 2)) + " (" + destPort + ")");
464                parameters.put("Source port", HexUtils.prettyHexString(HexUtils.integerToHexString(sourcePort, 2)) + " (" + sourcePort + ")");
465                parameters.put("Protocol", HexUtils.prettyHexString(HexUtils.integerToHexString(protocol.getID(), 1)) + " (" + protocol.getName() + ")");
466                parameters.put("RESTful Response Code", HexUtils.prettyHexString(HexUtils.integerToHexString(restFulStatus.getID(), 2)) + " (" + restFulStatus.getDescription() + ")");
467                if (data != null)
468                        parameters.put("Data", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(data)));
469                return parameters;
470        }
