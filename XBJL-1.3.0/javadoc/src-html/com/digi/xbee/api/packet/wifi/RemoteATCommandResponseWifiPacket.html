035/**
036 * This class represents a Remote AT Command Response (Wi-Fi) packet. Packet is
037 * built using the parameters of the constructor or providing a valid API
038 * payload.
039 *
040 * <p>If a module receives a remote command response RF data frame in response
041 * to a Remote AT Command Request, the module will send a Remote AT Command
042 * Response message out the UART. Some commands may send back multiple frames--
043 * for example, Node Discover ({@code ND}) command.</p>
044 *
045 * <p>This packet is received in response of a {@code RemoteATCommandPacket}.
046 * </p>
047 *
048 * <p>Response also includes an {@code ATCommandStatus} object with the status
049 * of the AT command.</p>
050 *
051 * @see RemoteATCommandWifiPacket
052 * @see com.digi.xbee.api.models.ATCommandStatus
053 * @see com.digi.xbee.api.packet.XBeeAPIPacket
054 * 
055 * @since 1.2.0
056 */
057public class RemoteATCommandResponseWifiPacket extends XBeeAPIPacket {
058
059        // Constants.
060        private static final int MIN_API_PAYLOAD_LENGTH = 13; // 1 (Frame type) + 1 (frame ID) + 8 (IP address) + 2 (AT command) + 1 (status)
061
062        private static final String ERROR_PAYLOAD_NULL = "Remote AT Command Response (Wi-Fi) packet payload cannot be null.";
063        private static final String ERROR_INCOMPLETE_PACKET = "Incomplete Remote AT Command Response (Wi-Fi) packet.";
064        private static final String ERROR_NOT_VALID_PACKET = "Payload is not a Remote AT Command Response (Wi-Fi) packet.";
065        private static final String ERROR_FRAME_ID_ILLEGAL = "Frame ID must be between 0 and 255.";
066        private static final String ERROR_SOURCE_ADDRESS_NULL = "Source address cannot be null.";
067        private static final String ERROR_AT_COMMAND_NULL = "AT command cannot be null.";
068        private static final String ERROR_STATUS_NULL = "AT command status cannot be null.";
069
070        // Variables.
071        private final Inet4Address sourceAddress;
072
073        private final ATCommandStatus status;
074
075        private final String command;
076
077        private byte[] commandValue;
078
079        private Logger logger;
080
081        /**
082         * Creates an new {@code RemoteATCommandResponseWifiPacket} object from the
083         * given payload.
084         *
085         * @param payload The API frame payload. It must start with the frame type
086         *                corresponding to a Remote AT Command Response (Wi-Fi)
087         *                packet ({@code 0x87}). The byte array must be in
088         *                {@code OperatingMode.API} mode.
089         *
090         * @return Parsed Remote AT Command Response (Wi-Fi) packet.
091         *
092         * @throws IllegalArgumentException if {@code payload[0] != APIFrameType.REMOTE_AT_COMMAND_RESPONSE_WIFI.getValue()} or
093         *                                  if {@code payload.length < }{@value #MIN_API_PAYLOAD_LENGTH}.
094         * @throws NullPointerException if {@code payload == null}.
095         */
096        public static RemoteATCommandResponseWifiPacket createPacket(byte[] payload) {
097                if (payload == null)
098                        throw new NullPointerException(ERROR_PAYLOAD_NULL);
099
100                if (payload.length < MIN_API_PAYLOAD_LENGTH)
101                        throw new IllegalArgumentException(ERROR_INCOMPLETE_PACKET);
102
103                if ((payload[0] & 0xFF) != APIFrameType.REMOTE_AT_COMMAND_RESPONSE_WIFI.getValue())
104                        throw new IllegalArgumentException(ERROR_NOT_VALID_PACKET);
105
106                // payload[0] is the frame type.
107                int index = 1;
108
109                // Frame ID byte.
110                int frameID = payload[index] & 0xFF;
111                index = index + 1;
112
113                // 8 bytes of source address.
114                Inet4Address sourceAddress;
115                try {
116                        sourceAddress = (Inet4Address) Inet4Address.getByAddress(Arrays.copyOfRange(payload, index + 4, index + 8));
117                } catch (UnknownHostException e) {
118                        throw new IllegalArgumentException(e);
119                }
120                index = index + 8;
121
122                // 2 bytes of AT command.
123                String command = new String(new byte[]{payload[index], payload[index + 1]});
124                index = index + 2;
125
126                // Status byte.
127                int status = payload[index] & 0xFF;
128                index = index + 1;
129
130                // Get data.
131                byte[] commandData = null;
132                if (index < payload.length)
133                        commandData = Arrays.copyOfRange(payload, index, payload.length);
134
135                return new RemoteATCommandResponseWifiPacket(frameID, sourceAddress, command,
136                                ATCommandStatus.get(status), commandData);
137        }
138
139        /**
140         * Class constructor. Instantiates a new
141         * {@code RemoteATCommandResponseWifiPacket} object with the given parameters.
142         *
143         * @param frameID frame ID.
144         * @param sourceAddress IP address of the remote radio returning
145         *                      response.
146         * @param command The AT command.
147         * @param status The command status.
148         * @param commandValue The AT command response value.
149         *
150         * @throws IllegalArgumentException if {@code frameID < 0} or
151         *                                  if {@code frameID > 255}.
152         * @throws NullPointerException if {@code sourceAddress == null} or
153         *                              if {@code command == null} or
154         *                              if {@code status == null}.
155         *
156         * @see com.digi.xbee.api.models.ATCommandStatus
157         * @see com.digi.xbee.api.models.XBee16BitAddress
158         * @see com.digi.xbee.api.models.XBee64BitAddress
159         * @see java.net.Inet4Address
160         */
161        public RemoteATCommandResponseWifiPacket(int frameID, Inet4Address sourceAddress,
162                        String command, ATCommandStatus status, byte[] commandValue) {
163                super(APIFrameType.REMOTE_AT_COMMAND_RESPONSE_WIFI);
164
165                if (frameID < 0 || frameID > 255)
166                        throw new IllegalArgumentException(ERROR_FRAME_ID_ILLEGAL);
167                if (sourceAddress == null)
168                        throw new NullPointerException(ERROR_SOURCE_ADDRESS_NULL);
169                if (command == null)
170                        throw new NullPointerException(ERROR_AT_COMMAND_NULL);
171                if (status == null)
172                        throw new NullPointerException(ERROR_STATUS_NULL);
173
174                this.frameID = frameID;
175                this.sourceAddress = sourceAddress;
176                this.command = command;
177                this.status = status;
178                this.commandValue = commandValue;
179                this.logger = LoggerFactory.getLogger(RemoteATCommandResponseWifiPacket.class);
180        }
181
182        /*
183         * (non-Javadoc)
184         * @see com.digi.xbee.api.packet.XBeeAPIPacket#getAPIPacketSpecificData()
185         */
186        @Override
187        protected byte[] getAPIPacketSpecificData() {
188                ByteArrayOutputStream os = new ByteArrayOutputStream();
189                try {
190                        os.write(new byte[]{0x00, 0x00, 0x00, 0x00}); // First 4 bytes of the 64-bit source address.
191                        os.write(sourceAddress.getAddress());
192                        os.write(ByteUtils.stringToByteArray(command));
193                        os.write(status.getId());
194                        if (commandValue != null)
195                                os.write(commandValue);
196                } catch (IOException e) {
197                        logger.error(e.getMessage(), e);
198                }
199                return os.toByteArray();
200        }

290                return commandValue;

292
293        /**
294         * Returns the AT command response value as String.
295         *
296         * @return The AT command response value as String, {@code null} if no
297         *         value is set.
298         *
299         * @see #getCommandValue()
300         * @see #setCommandValue(byte[])
301         * @see #setCommandValue(String)
302         */
303        public String getCommandValueAsString() {
304                if (commandValue == null)
305                        return null;
306                return new String(commandValue);
307        }
308
309        /*
310         * (non-Javadoc)
311         * @see com.digi.xbee.api.packet.XBeeAPIPacket#getAPIPacketParameters()
312         */
313        @Override
314        public LinkedHashMap<String, String> getAPIPacketParameters() {
315                LinkedHashMap<String, String> parameters = new LinkedHashMap<String, String>();
316                parameters.put("Source address", "00 00 00 00 " + HexUtils.prettyHexString(HexUtils.byteArrayToHexString(sourceAddress.getAddress())) + " (" + sourceAddress.getHostAddress() + ")");
317                parameters.put("AT Command", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(command.getBytes())) + " (" + command + ")");
318                parameters.put("Status", HexUtils.prettyHexString(HexUtils.integerToHexString(status.getId(), 1)) + " (" + status.getDescription() + ")");
319                if (commandValue != null) {
320                        if (ATStringCommands.get(command) != null)
321                                parameters.put("Response", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(commandValue)) + " (" + new String(commandValue) + ")");
322                        else
323                                parameters.put("Response", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(commandValue)));
324                }
325                return parameters;
326        }
