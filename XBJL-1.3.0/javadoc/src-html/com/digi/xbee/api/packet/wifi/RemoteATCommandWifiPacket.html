034/**
035 * This class represents a Remote AT Command Request (Wi-Fi) packet. Packet is
036 * built using the parameters of the constructor or providing a valid API
037 * payload.
038 *
039 * <p>Used to query or set module parameters on a remote device. For parameter
040 * changes on the remote device to take effect, changes must be applied, either
041 * by setting the apply changes options bit, or by sending an {@code AC} command
042 * to the remote node.</p>
043 *
044 * <p>Remote Command options are set as a bitfield.</p>
045 *
046 * <p>If configured, command response is received as a
047 * {@code RemoteATCommandResponseWifi packet}.</p>
048 *
049 * @see RemoteATCommandResponseWifiPacket
050 * @see com.digi.xbee.api.packet.XBeeAPIPacket
051 * 
052 * @since 1.2.0
053 */
054public class RemoteATCommandWifiPacket extends XBeeAPIPacket {
055
056        // Constants.
057        private static final int MIN_API_PAYLOAD_LENGTH = 13; // 1 (Frame type) + 1 (frame ID) + 8 (IP address) + 1 (transmit options byte) + 2 (AT command)
058
059        private static final String ERROR_PAYLOAD_NULL = "Remote AT Command Request (Wi-Fi) packet payload cannot be null.";
060        private static final String ERROR_INCOMPLETE_PACKET = "Incomplete Remote AT Command Request (Wi-Fi) packet.";
061        private static final String ERROR_NOT_VALID_PACKET = "Payload is not a Remote AT Command Request (Wi-Fi) packet.";
062        private static final String ERROR_FRAME_ID_ILLEGAL = "Frame ID must be between 0 and 255.";
063        private static final String ERROR_DEST_ADDRESS_NULL = "Destination address cannot be null.";
064        private static final String ERROR_OPTIONS_INVALID = "Receive options must be between 0 and 255.";
065        private static final String ERROR_AT_COMMAND_NULL = "AT command cannot be null.";
066
067        // Variables.
068        private final Inet4Address destAddress;
069
070        private final int transmitOptions;
071
072        private final String command;
073
074        private byte[] parameter;
075
076        private Logger logger;
077
078        /**
079         * Creates a new {@code RemoteATCommandWifiPacket} object from the given
080         * payload.
081         *
082         * @param payload The API frame payload. It must start with the frame type
083         *                corresponding to a Remote AT Command Requst (Wi-Fi) packet
084         *                ({@code 0x07}). The byte array must be in
085         *                {@code OperatingMode.API} mode.
086         *
087         * @return Parsed Remote AT Command Request (Wi-Fi) packet.
088         *
089         * @throws IllegalArgumentException if {@code payload[0] != APIFrameType.REMOTE_AT_COMMAND_REQUEST_WIFI.getValue()} or
090         *                                  if {@code payload.length < }{@value #MIN_API_PAYLOAD_LENGTH}.
091         * @throws NullPointerException if {@code payload == null}.
092         */
093        public static RemoteATCommandWifiPacket createPacket(byte[] payload) {
094                if (payload == null)
095                        throw new NullPointerException(ERROR_PAYLOAD_NULL);
096
097                if (payload.length < MIN_API_PAYLOAD_LENGTH)
098                        throw new IllegalArgumentException(ERROR_INCOMPLETE_PACKET);
099
100                if ((payload[0] & 0xFF) != APIFrameType.REMOTE_AT_COMMAND_REQUEST_WIFI.getValue())
101                        throw new IllegalArgumentException(ERROR_NOT_VALID_PACKET);
102
103                // payload[0] is the frame type.
104                int index = 1;
105
106                // Frame ID byte.
107                int frameID = payload[index] & 0xFF;
108                index = index + 1;
109
110                // 8 bytes of destination address.
111                Inet4Address destAddress;
112                try {
113                        destAddress = (Inet4Address) Inet4Address.getByAddress(Arrays.copyOfRange(payload, index + 4, index + 8));
114                } catch (UnknownHostException e) {
115                        throw new IllegalArgumentException(e);
116                }
117                index = index + 8;
118
119                // Options byte.
120                int transmitOptions = payload[index] & 0xFF;
121                index = index + 1;
122
123                // 2 bytes of AT command.
124                String command = new String(new byte[]{payload[index], payload[index + 1]});
125                index = index + 2;
126
127                // Get data.
128                byte[] parameterData = null;
129                if (index < payload.length)
130                        parameterData = Arrays.copyOfRange(payload, index, payload.length);
131
132                return new RemoteATCommandWifiPacket(frameID, destAddress, transmitOptions, command, parameterData);
133        }
134
135        /**
136         * Class constructor. Instantiates a new {@code RemoteATCommandWifiPacket}
137         * object with the given parameters.
138         *
139         * @param frameID The Frame ID.
140         * @param destAddress IP address of the destination device.
141         * @param transmitOptions Bitfield of supported transmission options.
142         * @param command AT command.
143         * @param parameter AT command parameter as byte array.
144         *
145         * @throws IllegalArgumentException if {@code frameID < 0} or
146         *                                  if {@code frameID > 255} or
147         *                                  if {@code transmitOptions < 0} or
148         *                                  if {@code transmitOptions > 255}.
149         * @throws NullPointerException if {@code destAddress == null} or
150         *                              if {@code command == null}.
151         *
152         * @see #RemoteATCommandWifiPacket(int, Inet4Address, int, String, String)
153         * @see com.digi.xbee.api.models.RemoteATCommandOptions
154         * @see java.net.Inet4Address
155         */
156        public RemoteATCommandWifiPacket(int frameID, Inet4Address destAddress, int transmitOptions,
157                        String command, byte[] parameter) {
158                super(APIFrameType.REMOTE_AT_COMMAND_REQUEST_WIFI);
159
160                if (frameID < 0 || frameID > 255)
161                        throw new IllegalArgumentException(ERROR_FRAME_ID_ILLEGAL);
162                if (destAddress == null)
163                        throw new NullPointerException(ERROR_DEST_ADDRESS_NULL);
164                if (transmitOptions < 0 || transmitOptions > 255)
165                        throw new IllegalArgumentException(ERROR_OPTIONS_INVALID);
166                if (command == null)
167                        throw new NullPointerException(ERROR_AT_COMMAND_NULL);
168
169                this.frameID = frameID;
170                this.destAddress = destAddress;
171                this.transmitOptions = transmitOptions;
172                this.command = command;
173                this.parameter = parameter;
174                this.logger = LoggerFactory.getLogger(RemoteATCommandWifiPacket.class);
175        }
176
177        /**
178         * Class constructor. Instantiates a new {@code RemoteATCommandWifiPacket}
179         * object with the given parameters.
180         *
181         * @param frameID The Frame ID.
182         * @param destAddress IP address of the destination device.
183         * @param transmitOptions Bitfield of supported transmission options.
184         * @param command AT command.
185         * @param parameter AT command parameter as string.
186         *
187         * @throws IllegalArgumentException if {@code frameID < 0} or
188         *                                  if {@code frameID > 255} or
189         *                                  if {@code transmitOptions < 0} or
190         *                                  if {@code transmitOptions > 255}.
191         * @throws NullPointerException if {@code destAddress == null} or
192         *                              if {@code command == null}.
193         *
194         * @see #RemoteATCommandWifiPacket(int, Inet4Address, int, String, byte[])
195         * @see com.digi.xbee.api.models.RemoteATCommandOptions
196         * @see java.net.Inet4Address
197         */
198        public RemoteATCommandWifiPacket(int frameID, Inet4Address destAddress, int transmitOptions,
199                        String command, String parameter) {
200                this(frameID, destAddress, transmitOptions, command, parameter == null ? null : parameter.getBytes());
201        }
202
203        /*
204         * (non-Javadoc)
205         * @see com.digi.xbee.api.packet.XBeeAPIPacket#getAPIPacketSpecificData()
206         */
207        @Override
208        protected byte[] getAPIPacketSpecificData() {
209                ByteArrayOutputStream os = new ByteArrayOutputStream();
210                try {
211                        os.write(new byte[]{0x00, 0x00, 0x00, 0x00}); // First 4 bytes of the 64-bit source address.
212                        os.write(destAddress.getAddress());
213                        os.write(transmitOptions);
214                        os.write(command.getBytes());
215                        if (parameter != null)
216                                os.write(parameter);
217                } catch (IOException e) {
218                        logger.error(e.getMessage(), e);
219                }
220                return os.toByteArray();
221        }

239        }

281        public void setParameter(String parameter) {
282                if (parameter == null)
283                        this.parameter = null;
284                else
285                        this.parameter = parameter.getBytes();
286        }

296         */
297        public void setParameter(byte[] parameter) {
298                this.parameter = parameter;
299        }
300
301        /**
302         * Returns the AT command parameter.
303         *
304         * @return The AT command parameter.
305         *
306         * @see #getParameterAsString()
307         * @see #setParameter(byte[])
308         * @see #setParameter(String)
309         */
310        public byte[] getParameter() {
311                return parameter;
312        }
313
314        /**
315         * Returns the AT command parameter as String.
316         *
317         * @return The AT command parameter as String, {@code null} if it is not
318         *         set.
319         *
320         * @see #getParameter()
321         * @see #setParameter(byte[])
322         * @see #setParameter(String)
323         */
324        public String getParameterAsString() {
325                if (parameter == null)
326                        return null;
327                return new String(parameter);
328        }
329
330        /*
331         * (non-Javadoc)
332         * @see com.digi.xbee.packet.XBeeAPIPacket#getAPIPacketParameters()
333         */
334        @Override
335        public LinkedHashMap<String, String> getAPIPacketParameters() {
336                LinkedHashMap<String, String> parameters = new LinkedHashMap<String, String>();
337                parameters.put("Destination address", "00 00 00 00 " + HexUtils.prettyHexString(HexUtils.byteArrayToHexString(destAddress.getAddress())) + " (" + destAddress.getHostAddress() + ")");
338                parameters.put("Transmit options", HexUtils.prettyHexString(HexUtils.integerToHexString(transmitOptions, 1)));
339                parameters.put("AT Command", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(command.getBytes())) + " (" + command + ")");
340                if (parameter != null) {
341                        if (ATStringCommands.get(command) != null)
342                                parameters.put("Parameter", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(parameter)) + " (" + new String(parameter) + ")");
343                        else
344                                parameters.put("Parameter", HexUtils.prettyHexString(HexUtils.byteArrayToHexString(parameter)));
345                }
346                return parameters;
347        }
