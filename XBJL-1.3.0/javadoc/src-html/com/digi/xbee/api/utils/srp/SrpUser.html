019package com.digi.xbee.api.utils.srp;
020
021import java.io.ByteArrayOutputStream;
022import java.io.IOException;
023import java.math.BigInteger;
024import java.security.MessageDigest;
025import java.security.NoSuchAlgorithmException;
026import java.security.SecureRandom;
027
028/**
029 * This class represents the user role in the Secure Remote Password protocol.
030 *
031 * @since 1.3.0
032 */
033public class SrpUser {
034
035        private String username;
036        private String password;
037
038        private boolean authenticated = false;
039
040        private BigInteger N;
041        private BigInteger g;
042        private BigInteger a;
043        private byte[] byte_A;
044        private BigInteger k;
045        private byte[] H_AMK;
046        private byte[] byte_K;
047
048        /**
049         * Class constructor. Instantiates a new {@code SrpUser} object with the
050         * given parameters.
051         *
052         * @param username The username to authenticate.
053         * @param password The password to authenticate.
054         */
055        public SrpUser(String username, String password) {
056                this(username, password, null, null);
057        }
058
059        /**
060         * Class constructor. Instantiates a new {@code SrpUser} object with the
061         * given parameters.
062         *
063         * @param username The username to authenticate.
064         * @param password The password to authenticate.
065         * @param byte_N N as per the SRP formula. If not passed, a default 1024 bit
066         *               N is used.
067         * @param byte_g The corresponding generator value for N, by default, 2.
068         */
069        private SrpUser(String username, String password, byte[] byte_N, byte[] byte_g) {
070                this.username = username;
071                this.password = password;
072                N = byte_N != null ? SrpUtils.bigIntegerFromBytes(byte_N) : SrpConstants.N;
073                g = byte_g != null ? SrpUtils.bigIntegerFromBytes(byte_g) : SrpConstants.g;
074        }
075
076        /**
077         * Computes and returns A, which is needed to start authentication.
078         *
079         * @return Public ephemeral value A.
080         * 
081         * @throws IOException If there is a problem generating 'byte k' value.
082         * @throws NoSuchAlgorithmException If the hash algorithm to use does not
083         *                                  exist.
084         */
085        public byte[] startAuthentication() throws NoSuchAlgorithmException, IOException {
086                // Reset variables.
087                authenticated = false;
088                byte_K = null;
089
090                // Generate a random 32 byte a.
091                byte[] byte_a = new byte[32];
092                new SecureRandom().nextBytes(byte_a);
093                this.a = SrpUtils.bigIntegerFromBytes(byte_a);
094
095                // Compute A.
096                byte_A = SrpUtils.bigIntegerToBytes(g.modPow(a, N));
097
098                // Compute k in byte array form.
099                byte[] byte_k;
100                MessageDigest digest = MessageDigest.getInstance(SrpConstants.HASH_ALGORITHM);
101                ByteArrayOutputStream os = new ByteArrayOutputStream();
102                os.write(SrpUtils.bigIntegerToBytes(N));
103                os.write(SrpUtils.bigIntegerToBytes(g));
104                byte_k = digest.digest(os.toByteArray());
105
106                // Get BigInteger k and store it.
107                this.k = SrpUtils.bigIntegerFromBytes(byte_k);
108
109                return byte_A;
110        }
111
112        /**
113         * Returns M1 if the challenge was successfully processed.
114         *
115         * @param byte_s Salt.
116         * @param byte_B Public ephemeral value B.
117         *
118         * @return Proof of session key M1, or {@code null} if the challenge could
119         *         not be processed.
120         * 
121         * @throws IOException If there is a problem generating any value.
122         * @throws NoSuchAlgorithmException If the hash algorithm to use does not
123         *                                  exist.
124         */
125        public byte[] processChallenge(byte[] byte_s, byte[] byte_B) throws NoSuchAlgorithmException, IOException {
126                BigInteger B = SrpUtils.bigIntegerFromBytes(byte_B);
127
128                // SRP-6a dictated safety check.
129                if (B.mod(N).equals(BigInteger.ZERO))
130                        return null;
131
132                // Compute M.
133                byte[] byte_M;
134
135                MessageDigest digest = MessageDigest.getInstance(SrpConstants.HASH_ALGORITHM);
136                ByteArrayOutputStream os = new ByteArrayOutputStream();
137                os.write(byte_A);
138                os.write(SrpUtils.bigIntegerToBytes(B));
139                byte[] byte_u = digest.digest(os.toByteArray());
140
141                BigInteger u = SrpUtils.bigIntegerFromBytes(byte_u);
142
143                // SRP-6a dictated safety check.
144                if (u.equals(BigInteger.ZERO))
145                        return null;
146
147                // Compute x.
148                byte[] byte_I = username.getBytes();
149                byte[] byte_p = password.getBytes();
150                byte[] byte_x = SrpUtils.generateX(byte_s, byte_I, byte_p);
151                BigInteger x = SrpUtils.bigIntegerFromBytes(byte_x);
152
153                // Compute v.
154                BigInteger v = g.modPow(x, N).abs();
155
156                // Compute S.
157                // The remainder is computed here, not the modulo.
158                // This means that, if n is negative, we need to do N - remainder to get the modulo.
159                BigInteger S = B.subtract(k.multiply(v)).modPow(a.add(u.multiply(x)), N);
160
161                if (S.compareTo(BigInteger.ZERO) < 0)
162                        S = N.add(S);
163
164                // Compute K.
165                byte_K = digest.digest(SrpUtils.bigIntegerToBytes(S));
166
167                // Compute M.
168                byte_M = SrpUtils.generateM(SrpUtils.bigIntegerToBytes(N), SrpUtils.bigIntegerToBytes(g), byte_I,
169                                byte_s, byte_A, SrpUtils.bigIntegerToBytes(B), byte_K);
170
171                // And finally, hash A, M and K together.
172                os = new ByteArrayOutputStream();
173                os.write(byte_A);
174                os.write(byte_M);
175                os.write(byte_K);
176                H_AMK = digest.digest(os.toByteArray());
177
178                return byte_M;
179        }
180
181        /**
182         * Verifies the given server proof of session key M2.
183         *
184         * @param host_H_AMK Proof of session key M2.
185         */
186        public void verifySession(byte[] host_H_AMK) {
187                if (host_H_AMK.length != H_AMK.length)
188                        return;
189
190                for (int i = 0; i < H_AMK.length; i++) {
191                        if (H_AMK[i] != host_H_AMK[i])
192                                return;
193                }
194
195                authenticated = true;
196        }
197
198        /**
199         * Returns whether the user is authenticated or not.
200         *
201         * @return {@code true} if the user is authenticated, {@code false}
202         *         otherwise.
203         */
204        public boolean isAuthenticated() {
205                return authenticated;
206        }
207
208        /**
209         * Returns the session key.
210         *
211         * @return The session key, or {@code null} if it is not known yet.
212         */
213        public byte[] getSessionKey() {
214                return byte_K;
215        }
